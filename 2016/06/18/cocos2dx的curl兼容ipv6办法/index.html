cocos2dx的curl兼容ipv6办法

By kenithcai

2016-06-18 Update date:2016-06-18

替换2dx3.11的curl库

3.11版本的链接:http://www.cocos2d-x.org/filedown/cocos2d-x-3.11.1.zip

直接下载最新的引擎。拿里面的关于curl的库跟源码对旧项目进行替换。

注意:旧版本的cocos2dx只有libhcurl.a，而新版本的是有libcurl.a, libssl.a, libcrypto.a, 三个都要加进去，而且还要加到项目的引用。

判定当前网络环境


enum TLocalIPStack {
    ELocalIPStack_None = 0,
    ELocalIPStack_IPv4 = 1,
    ELocalIPStack_IPv6 = 2,
    ELocalIPStack_Dual = 3,
};

// 判定当前网络环境
TLocalIPStack NetIPV::getIpv()
{
    const char* ipaddr = "www.baidu.com";

    TLocalIPStack type;

    struct addrinfo *answer, hint, *curr;

    memset(&hint, 0, sizeof(hint));
    hint.ai_family = AF_UNSPEC;
    hint.ai_socktype = SOCK_STREAM;

    // 处理名字到地址以及服务到端口这两种转换，返回的是一个sockaddr结构的链表
    int ret = getaddrinfo(ipaddr, NULL,&hint, &answer);

    if (ret != 0) type = ELocalIPStack_None;

    else
    {
        for (curr = answer; curr != NULL; curr = curr->ai_next) {
            switch (curr->ai_family)
            {
                case AF_UNSPEC:
                    //do something here
                    break;
                case AF_INET:
                    type = ELocalIPStack_IPv4;
                    break;
                case AF_INET6:
                    type = ELocalIPStack_IPv6;
                    break;
                default:
                    break;
            }
            if (type != ELocalIPStack_None) break;
        }
    }

    freeaddrinfo(answer);

    return type;
}

void NetIPV::replaceStr(std::string &s1, const std::string &s2, const std::string &s3)
{
    std::string::size_type pos=0;
    std::string::size_type a=s2.size();
    std::string::size_type b=s3.size();
    while((pos = s1.find(s2, pos)) != std::string::npos)
    {
        s1.replace(pos, a, s3);
        pos += b;
    }
}

http链接的转换

我们的连接是ipv4的连接地址,如:http://192.168.0.1/xxxxxx,http://192.168.0.1:80/xxxx


// 转换http链接
std::string NetIPV::getIpvUrl(const char *url)
{
    std::string tmp = url;

    TLocalIPStack type = NetIPV::getIpv();
    if (type == ELocalIPStack_IPv4)
    {
        tmp = url;
    }else if(type == ELocalIPStack_IPv6)
    {
        std::string oIp;
        // 分割链接
        std::vector list = NetIPV::split(url, "/");

        // 循环找出ip地址
        std::vector::iterator it = list.begin();
        for (; it != list.end(); ++it)
        {
            // 有可能有端口号,需要再分割一次
            std::vector tmpList = NetIPV::split(*it, ":");
            std::vector::iterator tmpIt = tmpList.begin();
            for (; tmpIt != tmpList.end(); ++tmpIt)
            {
                // 判断是否ip地址
                if (INADDR_NONE != inet_addr(tmpIt->c_str()))
                {
                    std::string::size_type pos = tmpIt->find(".");
                    if (pos != std::string::npos)
                    {
                        oIp = *tmpIt;
                        break;
                    }
                }
                if (oIp != "") break;
            }


        }
        // 将v4地址改成v6地址
        char nIp[256] = {0};
        sprintf(nIp, "[64:ff9b::%s]", oIp.c_str());

        // 替换地址中的v4地址改成v6
        NetIPV::replaceStr(tmp, oIp, nIp);
    }
    return tmp;
}
